SQL հրամաններ և օրինակներ

Սեղմիր հրամանի վրա՝ մանրամասները տեսնելու համար

SELECT հրամանը օգտագործվում է տվյալներ ընտրելու համար տվյալների բազայից: Կարող ենք ընտրել ամբողջ աղյուսակը կամ միայն որոշ սյունակներ: 

-- Բոլոր տողերը և սյունակները
SELECT * FROM Users;

-- Մոնորոշ սյունակներ
SELECT name, age FROM Users;

-- Հեռացում կրկնվող արժեքների
SELECT DISTINCT age FROM Users;

Օրինակ՝ դասավորել ըստ տարիքի:

SELECT name, age FROM Users ORDER BY age DESC;

INSERT հրամանը ավելացնում է նոր տվյալներ աղյուսակում:

INSERT INTO Users (name, age, city)
VALUES ('Անի', 20, 'Երևան');

-- Ավելացնել մի քանի տող
INSERT INTO Users (name, age, city)
VALUES 
('Մարի', 22, 'Գյումրի'),
('Արամ', 19, 'Երևան');

UPDATE-ը փոխում է գոյություն ունեցող տվյալները՝ ըստ որոշակի պայմանների: 

-- Թարմացնել Անիի տարիքը
UPDATE Users SET age=21 WHERE name='Անի';

-- Թարմացնել բոլոր օգտագործողների քաղաքը
UPDATE Users SET city='Վանաձոր' WHERE city='Գյումրի';

DELETE հրամանը ջնջում է տվյալներ ըստ պայմանների:

-- Ջնջել 18 տարեկանից փոքր օգտագործողներին
DELETE FROM Users WHERE age < 18;

-- Ջնջել բոլոր օգտագործողներին (Ուշադրություն!)
DELETE FROM Users;

Աղյուսակ ստեղծելու և փոփոխելու հրամաններ:

-- Աղյուսակ ստեղծել
CREATE TABLE Users (
  ID INT PRIMARY KEY,
  name VARCHAR(50),
  age INT,
  city VARCHAR(50)
);

-- Ավելացնել նոր սյունակ
ALTER TABLE Users ADD COLUMN email VARCHAR(100);

-- Փոխել սյունակի տեսակը
ALTER TABLE Users MODIFY COLUMN age SMALLINT;
-- Աղյուսակ ջնջել
DROP TABLE Users;

-- Սյունակ ջնջել
ALTER TABLE Users DROP COLUMN email;

WHERE-ով սահմանում ենք ֆիլտրի պայմաններ:

-- հավասարություն
SELECT * FROM Users WHERE city='Երևան';

-- բազմապատիկ պայմաններ
SELECT * FROM Users WHERE age>20 AND city='Երևան';

-- OR և NOT օպերատորներ
SELECT * FROM Users WHERE age<18 OR city='Գյումրի';
SELECT * FROM Users WHERE NOT age=20;
-- LIKE օգտագործում ենք տեքստային ֆիլտրի համար
SELECT * FROM Users WHERE name LIKE 'Ա%';

-- IN օգտագործում ենք արժեքների ցուցակի համար
SELECT * FROM Users WHERE city IN ('Երևան','Գյումրի');
-- Միջակայք
SELECT * FROM Users WHERE age BETWEEN 18 AND 25;

-- NULL արժեքներ
SELECT * FROM Users WHERE email IS NULL;

Հաշվարկներ և խմբավորումներ:

SELECT COUNT(*) FROM Users;         -- Հաշվել տողերի քանակը
SELECT AVG(age) FROM Users;         -- Միջին տարիքը
SELECT SUM(age) FROM Users;         -- Տարիքների գումարը
SELECT MIN(age) FROM Users;         -- Նվազագույն տարիքը
SELECT MAX(age) FROM Users;         -- Ամենամեծ տարիքը
-- Խմբավորել ըստ քաղաքի
SELECT city, COUNT(*) FROM Users GROUP BY city;

-- Խմբավորել եւ ֆիլտրել
SELECT city, COUNT(*) FROM Users 
GROUP BY city 
HAVING COUNT(*) > 1;
-- Դասավորել ըստ տարիքի
SELECT * FROM Users ORDER BY age ASC;

-- Դասավորել և սահմանափակել արդյունքը
SELECT * FROM Users ORDER BY age DESC LIMIT 5;

Միացնում ենք տարբեր աղյուսակներ:

-- INNER JOIN: միայն համընկնող տողերը
SELECT Users.name, Orders.amount
FROM Users
INNER JOIN Orders ON Users.ID = Orders.user_id;

-- LEFT JOIN: բոլոր Users, նույնիսկ առանց պատվերի
SELECT Users.name, Orders.amount
FROM Users
LEFT JOIN Orders ON Users.ID = Orders.user_id;

-- RIGHT JOIN: բոլոր Orders, նույնիսկ առանց օգտվողի
SELECT Users.name, Orders.amount
FROM Users
RIGHT JOIN Orders ON Users.ID = Orders.user_id;

-- FULL JOIN (եթե աջակցվում է)
SELECT Users.name, Orders.amount
FROM Users
FULL OUTER JOIN Orders ON Users.ID = Orders.user_id;
-- Թարմացնել մի քանի աղյուսակ և հաստատել
START TRANSACTION;
UPDATE Users SET age=25 WHERE name='Անի';
INSERT INTO Logs (action) VALUES ('Age updated');
COMMIT;

-- Հակադարձել (rollback)
START TRANSACTION;
DELETE FROM Users WHERE age<18;
ROLLBACK;
-- Ավելացնել ինդեքս արագ որոնումների համար
CREATE INDEX idx_name ON Users(name);

-- Ջնջել ինդեքս
DROP INDEX idx_name ON Users;
-- Ստեղծել դիտում
CREATE VIEW ActiveUsers AS
SELECT * FROM Users WHERE active=1;

-- Օգտագործել դիտումը
SELECT * FROM ActiveUsers;
-- PRIMARY KEY
CREATE TABLE Users (
  ID INT PRIMARY KEY,
  name VARCHAR(50)
);

-- FOREIGN KEY
CREATE TABLE Orders (
  ID INT PRIMARY KEY,
  user_id INT,
  FOREIGN KEY (user_id) REFERENCES Users(ID)
);
-- UNIQUE
CREATE TABLE Users (
  ID INT PRIMARY KEY,
  email VARCHAR(100) UNIQUE
);

-- CHECK
CREATE TABLE Users (
  ID INT PRIMARY KEY,
  age INT CHECK(age>0)
);
